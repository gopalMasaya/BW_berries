function stream(data) {
    var input = data.val();
    if (!input) return;
    
    var keys = Object.keys(input);
    
    // Clear previous data
    allData.dates = [];
    allData.sc1 = [];
    allData.sc2 = [];
    allData.scAvg = [];
    allData.ec = [];
    allData.temprature = [];
    allData.waterFlow = [];
    allData.waterDrain = [];
    allData.voltage = [];
    allData.rollingSumWaterFlow24h = [];
    allData.rollingSumWaterDrain24h = [];
    allData.sumWaterFlow24h = 0;
    allData.sumWaterDrain24h = 0;

    // Clear and repopulate dailyData
    dailyData = [];
    
    // Process all data points
    for (let i = 0; i < keys.length; i++) {
        let k = keys[i];
        
        // Skip data points marked for deletion
        if (k.startsWith('29042025')) continue;
        
        allData.dates.push(k);
        allData.ec.push(input[k].ec);
        allData.temprature.push(input[k].temp);
        allData.sc1.push(input[k].scale1);
        allData.sc2.push(input[k].scale2);
        allData.voltage.push(input[k].voltage);
        const waterFlow = (input[k].waterIn / 4 * 0.05);
        allData.waterFlow.push(waterFlow);
        const waterOut = (input[k].waterOut / 4 * 0.05) / 2;
        allData.waterDrain.push(waterOut);
        
        // Add to dailyData array
        dailyData.push({
            date: k,
            ec: input[k].ec,
            temperature: input[k].temp,
            scale1: input[k].scale1,
            scale2: input[k].scale2,
            voltage: input[k].voltage,
            waterFlow: waterFlow,
            waterDrain: waterOut
        });
    }

    // Calculate rolling sums
    for (let i = 0; i < allData.dates.length; i++) {
        const currentTime = parseDateTimeFromKey(allData.dates[i]);
        let sumFlow = 0;
        let sumDrain = 0;

        for (let j = i; j >= 0; j--) {
            const compareTime = parseDateTimeFromKey(allData.dates[j]);
            const diffHours = (currentTime - compareTime) / (1000 * 60 * 60);
            if (diffHours <= 24) {
                sumFlow += allData.waterFlow[j];
                sumDrain += allData.waterDrain[j];
            } else {
                break;
            }
        }

        allData.rollingSumWaterFlow24h.push(sumFlow);
        allData.rollingSumWaterDrain24h.push(sumDrain);
    }

    // Calculate averages
    for(let i = 0; i < allData.sc1.length; i++ ){
        let avg = (allData.sc1[i]+allData.sc2[i])/2;
        allData.scAvg.push(avg);
    }
    
    let last = allData.sc1.length - 1;
    updateDiv(1, allData.sc1[last], allData.sc2[last], allData.ec[last], allData.temprature[last], allData.voltage[last], allData.waterFlow[last], allData.waterDrain[last]);
    
    // Apply date filtering if enabled, otherwise show all data
    if (dateFilterApplied) {
        applyDateFilter();
    } else {
        showDataForLast24Hours();
    }
}

function exportToExcel(data, date) {
    try {
        console.log('Exporting data:', data.length, 'records for date:', date);
        
        if (!window.XLSX) {
            console.error('XLSX library not found! Make sure to include SheetJS library.');
            return;
        }
        
        const worksheet = XLSX.utils.json_to_sheet(data);
        const workbook = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(workbook, worksheet, 'DailyData');
        
        // Generate filename with date
        XLSX.writeFile(workbook, `Data_${date}.xlsx`);
        console.log(`Excel file Data_${date}.xlsx created successfully!`);
    } catch (error) {
        console.error('Error creating Excel file:', error);
    }
}

// Schedule export at midnight
function scheduleMidnightExport() {
    console.log("Setting up midnight export schedule...");
    const now = new Date();
    const midnight = new Date(now);
    midnight.setHours(24, 0, 0, 0); // Set to next midnight
    
    const timeToMidnight = midnight.getTime() - now.getTime();
    console.log('Time to midnight:', Math.round(timeToMidnight / 1000 / 60), 'minutes');
    
    setTimeout(() => {
        console.log('Midnight export triggered!');
        const dailyDataToExport = getDailyData();
        if (dailyDataToExport.length > 0) {
            const yesterday = new Date(now);
            yesterday.setDate(yesterday.getDate() - 1);
            const day = String(yesterday.getDate()).padStart(2, '0');
            const month = String(yesterday.getMonth() + 1).padStart(2, '0');
            const year = yesterday.getFullYear();
            const dateStr = `${day}${month}${year}`;
            exportToExcel(dailyDataToExport, dateStr);
        } else {
            console.log('No data to export for yesterday');
        }
        
        // Schedule for next midnight
        setInterval(() => {
            console.log('Daily export triggered!');
            const newDailyData = getDailyData();
            if (newDailyData.length > 0) {
                const yesterday = new Date();
                yesterday.setDate(yesterday.getDate() - 1);
                const day = String(yesterday.getDate()).padStart(2, '0');
                const month = String(yesterday.getMonth() + 1).padStart(2, '0');
                const year = yesterday.getFullYear();
                const newDateStr = `${day}${month}${year}`;
                exportToExcel(newDailyData, newDateStr);
            } else {
                console.log('No data to export for yesterday');
            }
        }, 24 * 60 * 60 * 1000); // 24 hours
    }, timeToMidnight);
}

      function parseDateTimeFromKey(key) {
    const match = key.match(/^(\d{2})(\d{2})(\d{4})_(\d{2})(\d{2})(\d{2})$/);
    if (match) {
        const [_, dd, mm, yyyy, HH, MM, SS] = match;
        return new Date(`${yyyy}-${mm}-${dd}T${HH}:${MM}:${SS}`);
    }
    return new Date(0); // fallback to epoch if format is invalid
}

        // Chart Creation Functions
    function createChart(canvasId, label, color, yAxisLabel = '') {
    const ctx = document.getElementById(canvasId).getContext('2d');
    const chart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: [],
            datasets: [{
                label: label,
                data: [],
                borderColor: color,
                backgroundColor: color.replace('1)', '0.1)'),
                borderWidth: 3,
                pointRadius: 2,
                pointHoverRadius: 6,
                tension: 0.1,
                fill: true
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            animation: {
                duration: 300
            },
            interaction: {
                intersect: false,
                mode: 'index'
            },
            scales: {
                x: {
                    grid: {
                        color: 'rgba(0, 0, 0, 0.1)'
                    },
                    ticks: {
                        color: "#666",
                        maxRotation: 45,
                        minRotation: 45,
                        autoSkip: true,
                        maxTicksLimit: 10
                    }
                },
                y: {
                    grid: {
                        color: 'rgba(0, 0, 0, 0.1)'
                    },
                    ticks: {
                        color: "#666"
                    },
                    title: {
                        display: !!yAxisLabel,
                        text: yAxisLabel,
                        color: '#333'
                    }
                }
            },
            plugins: {
                legend: {
                    display: false
                },
                tooltip: {
                    backgroundColor: 'rgba(0, 0, 0, 0.8)',
                    titleColor: 'white',
                    bodyColor: 'white',
                    borderColor: color,
                    borderWidth: 2,
                    titleFont: {
                        size: 14
                    },
                    bodyFont: {
                        size: 14
                    },
                    padding: 12,
                    cornerRadius: 8
                }
            }
        }
    });

    // Add mouseenter event to show tooltip immediately
    chart.canvas.addEventListener('mouseenter', function(e) {
        if (chart.data.labels.length > 0) {
            // Create a synthetic mousemove event to trigger tooltip
            const rect = chart.canvas.getBoundingClientRect();
            const mouseEvent = new MouseEvent('mousemove', {
                clientX: e.clientX,
                clientY: e.clientY,
                bubbles: true
            });
                     
            // Dispatch the event to trigger Chart.js's built-in tooltip logic
            chart.canvas.dispatchEvent(mouseEvent);
        }
    });

    return chart;
}

// New function to create combined chart with multiple datasets
function createCombinedChart(canvasId, datasets, yAxisLabel = '') {
    const ctx = document.getElementById(canvasId).getContext('2d');
    const chart = new Chart(ctx, {
        type: 'line', // Base type, but we'll override for individual datasets
        
        data: {
            labels: [],
            datasets: datasets.map((dataset, index) => ({
                label: dataset.label,
                data: [],
                type: index === 0 ? 'line' : 'bar', // First dataset = line, others = bar
                borderColor: dataset.color.replace(/[\d.]+\)$/, '1)'), // force full opacity for border


                backgroundColor: index === 0 ? 
                     dataset.color.replace('1)', '0.1)') : 
                    dataset.color.replace('1)', '0.7)'), // More opacity for bars
                borderWidth: index === 0 ? 2 : 1, // Thicker line for line chart
                pointRadius: index === 0 ? 2 : 0, // Points only for line chart
                pointHoverRadius: index === 0 ? 8 : 0,
                tension: index === 0 ? 0.1 : 0, // Smooth line only for line chart
                fill: false,
                // Add units for value labels
                units: index === 0 ? 'nm/c' : (index === 1 ? 'ml' : 'ml')
            }))
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            animation: {
                duration: 300
            },
            interaction: {
                intersect: false,
                mode: 'index'
            },
            scales: {
                x: {
                    grid: {
                        color: 'rgba(0, 0, 0, 0.1)'
                    },
                    ticks: {
                        color: "#666",
                        maxRotation: 45,
                        minRotation: 45,
                        autoSkip: true,
                        maxTicksLimit: 10
                    }
                },
                y: {
                    grid: {
                        color: 'rgba(0, 0, 0, 0.1)'
                    },
                    ticks: {
                        color: "#666"
                    },
                    title: {
                        display: !!yAxisLabel,
                        text: yAxisLabel,
                        color: '#333'
                    }
                }
            },
            plugins: {
                legend: {
                    display: true,
                    position: 'top',
                    labels: {
                        color: '#333',
                        usePointStyle: true,
                        padding: 20
                    }
                },
                tooltip: {
                    backgroundColor: 'rgba(0, 0, 0, 0.8)',
                    titleColor: 'white',
                    bodyColor: 'white',
                    borderColor: 'rgba(255, 255, 255, 0.8)',
                    borderWidth: 1,
                    titleFont: {
                        size: 14
                    },
                    bodyFont: {
                        size: 14
                    },
                    padding: 12,
                    cornerRadius: 8,
                    // Custom tooltip to show units
                    callbacks: {
                        label: function(context) {
                            const dataset = context.dataset;
                            const value = context.parsed.y;
                            const units = dataset.units || '';
                            return `${dataset.label}: ${value.toFixed(3)} ${units}`;
                        }
                    }
                },
                // Plugin to display values on chart
                datalabels: {
                    display: function(context) {
                        // Show labels for bar charts only (not line chart)
                        return context.dataset.type === 'bar';
                    },
                    anchor: 'end',
                    align: 'top',
                    offset: 4,
                    color: '#333',
                    font: {
                        size: 11,
                        weight: 'bold'
                    },
                    formatter: function(value, context) {
                        const units = context.dataset.units || '';
                        return `${value.toFixed(3)} ${units}`;
                    }
                }
            }
        },
        plugins: [{
            // Custom plugin to display values on bars and side labels for percentages
            id: 'valueLabels',
            afterDatasetsDraw: function(chart) {
                const ctx = chart.ctx;
                
               
            }
        }]
    });

   

    // Add mouseenter event to show tooltip immediately
    chart.canvas.addEventListener('mouseenter', function(e) {
        if (chart.data.labels.length > 0) {
            const rect = chart.canvas.getBoundingClientRect();
            const mouseEvent = new MouseEvent('mousemove', {
                clientX: e.clientX,
                clientY: e.clientY,
                bubbles: true
            });
                     
            chart.canvas.dispatchEvent(mouseEvent);
        }
    });

    return chart;
}
