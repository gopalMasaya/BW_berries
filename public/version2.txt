        function formatDateKey(dateStr) {

            }
            return dateStr;

        function applyDateFilter() {
            fromDate = document.getElementById('fromDate').value;
            toDate = document.getElementById('toDate').value;

            if (!fromDate || !toDate) {
                alert("Please select both 'From' and 'To' dates");
                return;
            }
            
            var fromDateFormatted = fromDate.replace(/-/g, '');
            var toDateFormatted = toDate.replace(/-/g, '');
            
            const filteredData = {
                dates: [],
                ec: [],
                temprature: [],
                sc1: [],
                sc2: [],
                waterFlow: [],
                waterDrain: []
            };
            
            for (let i = 0; i < allData.dates.length; i++) {
                var dateKey = allData.dates[i];
                const formattedDateKey = formatDateKey(dateKey);

    function createChart(canvasId, label, color, yAxisLabel = '') {
    const ctx = document.getElementById(canvasId).getContext('2d');
    const chart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: [],
            datasets: [{
                label: label,
                data: [],
                borderColor: color,
                backgroundColor: color.replace('1)', '0.1)'),
                borderWidth: 3,
                pointRadius: 2,
                pointHoverRadius: 6,
                tension: 0.1,
                fill: true
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            animation: {
                duration: 300
            },
            interaction: {
                intersect: false,
                mode: 'index'
            },
            scales: {
                x: {
                    grid: {
                        color: 'rgba(0, 0, 0, 0.1)'
                    },
                    ticks: {
                        color: "#666",
                        maxRotation: 45,
                        minRotation: 45,
                        autoSkip: true,
                        maxTicksLimit: 10
                    }
                },
                y: {
                    grid: {
                        color: 'rgba(0, 0, 0, 0.1)'
                    },
                    ticks: {
                        color: "#666"
                    },
                    title: {
                        display: !!yAxisLabel,
                        text: yAxisLabel,
                        color: '#333'
                    }
                }
            },
            plugins: {
                legend: {
                    display: false
                },
                tooltip: {
                    backgroundColor: 'rgba(0, 0, 0, 0.8)',
                    titleColor: 'white',
                    bodyColor: 'white',
                    borderColor: color,
                    borderWidth: 2,
                    titleFont: {
                        size: 14
                    },
                    bodyFont: {
                        size: 14
                    },
                    padding: 12,
                    cornerRadius: 8
                }
            }
        }
    });

    // Add mouseenter event to show tooltip immediately
    chart.canvas.addEventListener('mouseenter', function(e) {
        if (chart.data.labels.length > 0) {
            // Create a synthetic mousemove event to trigger tooltip
            const rect = chart.canvas.getBoundingClientRect();
            const mouseEvent = new MouseEvent('mousemove', {
                clientX: e.clientX,
                clientY: e.clientY,
                bubbles: true
            });
                     
            // Dispatch the event to trigger Chart.js's built-in tooltip logic
            chart.canvas.dispatchEvent(mouseEvent);
        }
    });

    return chart;
}

                    titleColor: 'white',
                    bodyColor: 'white',
                    borderColor: 'rgba(255, 255, 255, 0.8)',
                    borderWidth: 1,
                    titleFont: {
                        size: 14
                    },
                    bodyFont: {
                        size: 14
                    },
                    padding: 12,
                    cornerRadius: 8
                }
            }
        }
    });

    // Add mouseenter event to show tooltip immediately
    chart.canvas.addEventListener('mouseenter', function(e) {
        if (chart.data.labels.length > 0) {
            const rect = chart.canvas.getBoundingClientRect();
            const mouseEvent = new MouseEvent('mousemove', {
                clientX: e.clientX,
                clientY: e.clientY,
                bubbles: true
            });
                     
            chart.canvas.dispatchEvent(mouseEvent);
        }
    });

    return chart;
}

// Helper function to add single data point to combined chart
function addDataToCombinedChart(chart, label, ecValue, waterInputValue, waterDrainValue) {
    chart.data.labels.push(label);
    chart.data.datasets[0].data.push(ecValue);      // EC dataset
    chart.data.datasets[1].data.push(waterInputValue); // Water Input dataset
    chart.data.datasets[2].data.push(waterDrainValue); // Water Drain dataset
    console.log('Adding data - EC:', ecValue, 'Water Input:', waterInputValue, 'Water Drain:', waterDrainValue);
    chart.update();
}
