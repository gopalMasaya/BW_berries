  function statusFromAge(mins){

  // Strict timestamp parser: expects Arduino `timestamp` (epoch seconds)
  function extractTimestampISO(item){
    if (!item || typeof item !== "object") return null;

    if (item.timestamp != null) {
      const n = Number(item.timestamp);
      if (Number.isFinite(n) && n > 1000000000) return new Date(n * 1000).toISOString();
      const d = new Date(item.timestamp);
      if (isFinite(d.getTime())) return d.toISOString();
    }
    
    // Prefer serverTimestamp for freshness/online status (more reliable)
if (item.serverTimestamp != null) {
  const ms = Number(item.serverTimestamp);
  if (Number.isFinite(ms) && ms > 1000000000000) {
    return new Date(ms).toISOString(); // serverTimestamp is ms
  }
}

    // optional fallback
    if (item.sortKey != null) {
      const n = Number(item.sortKey);
      if (Number.isFinite(n) && n > 1000000000) {
        const ms = (n < 10000000000) ? n * 1000 : n;
        return new Date(ms).toISOString();
      }
    }
    return null;
  }

  function buildStationsGeoJSON(){
    const features = STATIONS.map(st => {
      const latest = latestByDevice[st.deviceId];
      const mins = latest ? ageMinutes(latest.tsISO) : Infinity;
      const s = statusFromAge(mins);
      return {
        type: "Feature",
        geometry: { type: "Point", coordinates: [st.lng, st.lat] },
        properties: {
          deviceId: st.deviceId,
          name: st.name,
          status: s.cls,
          lastSeen: latest ? latest.tsISO : null
        }
      };
    });
    return { type: "FeatureCollection", features };
  }

  function refreshStationsSource(){
    if (!map) return;
    const src = map.getSource("stations");
    if (src) src.setData(buildStationsGeoJSON());
  }

  async function fetchLatestTelemetry(deviceId){
    const url = `${API_BASE}/telemetry?deviceId=${encodeURIComponent(deviceId)}&limit=1`;
    const res = await fetch(url, {
      method: "GET",
      headers: { "x-api-key": API_KEY, "Accept": "application/json" }
    });

    const text = await res.text();
    let json;
    try { json = JSON.parse(text); } catch { json = { raw: text }; }

    if (!res.ok) throw new Error(`HTTP ${res.status} ${res.statusText} → ${text}`);

    // API returns array
   // const item = Array.isArray(json) && json.length ? json[0] : null;
    const item = Array.isArray(json) && json.length
  ? json[json.length - 1]   // ← newest item
  : null;

    // keep last known data if empty
    if (!item) return { updated:false, tsISO: latestByDevice[deviceId]?.tsISO || null };

    const tsISO = extractTimestampISO(item);
    if (!tsISO) {
      const prev = latestByDevice[deviceId]?.tsISO || null;
      latestByDevice[deviceId] = { tsISO: prev, item };
      return { updated:true, tsISO: prev };
    }

    latestByDevice[deviceId] = { tsISO, item };
    return { updated:true, tsISO };
  }

  function evaluateAndAlert(deviceId){
    const latest = latestByDevice[deviceId];
    const mins = latest ? ageMinutes(latest.tsISO) : Infinity;
    const s = statusFromAge(mins);

    const prev = lastStatusByDevice[deviceId];
    lastStatusByDevice[deviceId] = s.cls;

    if (s.cls === "bad" && (prev !== "bad" || prev == null)){
      const st = STATIONS.find(x => x.deviceId === deviceId);
      const name = st ? st.name : deviceId;
      const lastSeen = latest ? new Date(latest.tsISO).toLocaleString() : "unknown";
      showAlert(`<strong>${name}</strong> is <strong>OFFLINE</strong>. Last data: ${lastSeen}.`);
    }
  }

  async function refreshAllStations(){
    const filter = deviceIdEl.value.trim();
    const targets = filter ? STATIONS.filter(s => s.deviceId === filter) : STATIONS;

    if (targets.length === 0){
      setStatus(`⚠ No station matches deviceId="${filter}"`, false);
      return;
    }

    try{
      const results = await Promise.all(targets.map(st => fetchLatestTelemetry(st.deviceId)));

      refreshStationsSource();
      renderStationList();
      targets.forEach(st => evaluateAndAlert(st.deviceId));

      const updatedCount = results.filter(r => r.updated).length;
      setStatus(buildStatusText(`✅ Updated ${updatedCount}/${targets.length}`), true);

      if (drawerDeviceId) renderDrawer(drawerDeviceId);

    } catch(err){
      console.error(err);
      targets.forEach(st => evaluateAndAlert(st.deviceId));
      refreshStationsSource();
      renderStationList();
      if (drawerDeviceId) renderDrawer(drawerDeviceId);

      setStatus(buildStatusText(`❌ Fetch failed: ${err.message}`), false);
    }
  }

        </div>
        <div class="pill"><span class="dot ${s.cls}"></span>${s.label}</div>
      `;

      item.addEventListener("click", () => {
        map.flyTo({ center: [st.lng, st.lat], zoom: 14, speed: 0.9 });
        openDrawer(st.deviceId);
      });

      stationListEl.appendChild(item);
    });
  }

  function openDrawer(deviceId){
    drawerDeviceId = deviceId;
    drawer.classList.add("open");
    drawer.setAttribute("aria-hidden", "false");
    renderDrawer(deviceId);
  }

  function closeDrawer(){
    drawerDeviceId = null;
    drawer.classList.remove("open");
    drawer.setAttribute("aria-hidden", "true");
  }

  function renderDrawer(deviceId){
    const st = STATIONS.find(s => s.deviceId === deviceId);
    const latest = latestByDevice[deviceId];

    drawerTitle.textContent = st ? st.name : deviceId;

    const mins = latest ? ageMinutes(latest.tsISO) : Infinity;
    const s = statusFromAge(mins);
    const lastSeen = latest ? new Date(latest.tsISO).toLocaleString() : "—";
    drawerSubtitle.textContent = `deviceId: ${deviceId} • ${s.label} • last seen: ${lastSeen}`;

    drawerGrid.innerHTML = "";

    if (!latest || !latest.item){
      drawerGrid.innerHTML = `<div style="grid-column:1 / -1; padding:10px; color:var(--muted); border:1px solid var(--border); border-radius:14px; background:rgba(255,255,255,0.03);">
        No telemetry loaded yet.
      </div>`;
      return;
    }

    const item = latest.item;

    const keys = Object.keys(item)
      .filter(k => k !== "deviceId" && k !== "sortKey" && !k.toLowerCase().includes("timestamp"))
      .filter(k => typeof item[k] === "number" || (!isNaN(parseFloat(item[k])) && item[k] !== null && item[k] !== ""))
      .sort();

    if (keys.length === 0){
      drawerGrid.innerHTML = `<div style="grid-column:1 / -1; padding:10px; color:var(--muted); border:1px solid var(--border); border-radius:14px; background:rgba(255,255,255,0.03);">
        No numeric sensor fields found in telemetry item.
      </div>`;
      return;
    }

    keys.forEach(k => {
      const v = parseFloat(item[k]);
      const card = document.createElement("div");
      card.className = "card";
      card.innerHTML = `
        <div class="k">${fmtFieldName(k)}</div>
        <div class="v">${Number.isFinite(v) ? v.toFixed(decimalsFor(k)) : "—"}</div>
        <div class="u">${unitFor(k)}</div>
        <div class="t">${new Date(latest.tsISO).toLocaleTimeString()}</div>
      `;
      drawerGrid.appendChild(card);
    });

    btnFly.onclick = () => {
      if (!st) return;
      map.flyTo({ center: [st.lng, st.lat], zoom: 15, speed: 0.9 });
    };

    btnRoute.onclick = () => {
      if (!st) return;
      if (!userLocation){
        alert("User location not available (browser denied).");
        return;
      }
      const [flng, flat] = userLocation;
      const url = `https://www.google.com/maps/dir/?api=1&origin=${flat},${flng}&destination=${st.lat},${st.lng}&travelmode=driving`;
      window.open(url, "_blank");
    };
  }

  drawerClose.addEventListener("click", closeDrawer);

  // ===============================
  // SCHEDULER (aligned 5-min + 1-min offset)
  // ===============================
  function computeNextFetchMs(nowMs){
    const now = new Date(nowMs);
    const secOfHour = now.getMinutes() * 60 + now.getSeconds(); // 0..3599

    // Fetch at offset=60s after each 5-min boundary:
    // times in hour: 60, 360, 660, 960, ...
    const base = Math.floor((secOfHour - FETCH_OFFSET_SEC) / PERIOD_SEC) + 1;
    const nextSecOfHour = base * PERIOD_SEC + FETCH_OFFSET_SEC;

    const next = new Date(now);
    next.setMilliseconds(0);

    if (nextSecOfHour < 3600){
      next.setMinutes(0, 0, 0);
      next.setSeconds(nextSecOfHour, 0);
      return next.getTime();
    }

    // spill into next hour
    const spill = nextSecOfHour - 3600;
    next.setHours(next.getHours() + 1, 0, 0, 0);
    next.setSeconds(spill, 0);
    return next.getTime();
  }

  function formatCountdown(msLeft){
    const s = Math.max(0, Math.floor(msLeft / 1000));
    const m = Math.floor(s / 60);
    const r = s % 60;
    return `${m}:${String(r).padStart(2,"0")}`;
  }

  function buildStatusText(prefix){
    const now = new Date();
    const next = nextFetchAtMs ? new Date(nextFetchAtMs) : null;
    const left = next ? formatCountdown(nextFetchAtMs - Date.now()) : "—";
    const nextStr = next ? next.toLocaleTimeString() : "—";
    return `${prefix}\nNow: ${now.toLocaleTimeString()} • Next fetch: ${nextStr} (in ${left})`;
  }

  function scheduleNextFetch(){
    if (fetchTimer) clearTimeout(fetchTimer);
    const nowMs = Date.now();
    nextFetchAtMs = computeNextFetchMs(nowMs);
    const delay = Math.max(250, nextFetchAtMs - nowMs);

    fetchTimer = setTimeout(async () => {
      await refreshAllStations();
      scheduleNextFetch(); // schedule the next run
    }, delay);
  }

  function startStatusTicker(){
    if (statusTickTimer) clearInterval(statusTickTimer);
    statusTickTimer = setInterval(() => {
      // Keep offline evaluation "live" between fetches
      STATIONS.forEach(st => evaluateAndAlert(st.deviceId));
      refreshStationsSource();
      renderStationList();
      if (drawerDeviceId) renderDrawer(drawerDeviceId);

      // Update status line (countdown)
      setStatus(buildStatusText("✅ Live (scheduled)"), true);
    }, TICK_SEC * 1000);
  }

  // ===============================
  // BOOT
  // ===============================
  async function boot(){
    if (navigator.geolocation){
      navigator.geolocation.getCurrentPosition(
        (pos) => { userLocation = [pos.coords.longitude, pos.coords.latitude]; },
        () => {},
        { enableHighAccuracy: true, timeout: 4000 }
      );
    }

    const center = [STATIONS[0].lng, STATIONS[0].lat];

    if (!MAPBOX_TOKEN || MAPBOX_TOKEN === "YOUR_MAPBOX_TOKEN_HERE"){
      setStatus("❌ Mapbox token missing. Set MAPBOX_TOKEN in code.", false);
      document.getElementById("map").innerHTML =
        `<div style="padding:18px;color:rgba(231,238,252,0.75)">
          <h3 style="margin:0 0 8px 0;">Map disabled</h3>
          <div style="color:rgba(231,238,252,0.55);line-height:1.4">
            Add your Mapbox token in <b>MAPBOX_TOKEN</b> to enable the map.
          </div>
        </div>`;
    } else {
      initMap(center);
    }

    renderStationList();

    // Do one initial fetch (so you see data immediately)
    await refreshAllStations();

    // Start the aligned scheduler (fetch at :01,:06,:11...)
    scheduleNextFetch();

    // Keep the UI + offline state fresh between fetches
    startStatusTicker();

    setStatus(buildStatusText("✅ Live (scheduled)"), true);
  }

  boot();
